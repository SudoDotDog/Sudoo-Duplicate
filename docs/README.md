## Install

```sh
npm install @sudoo/duplicate --save
# Or
yarn add @sudoo/duplicate
```

## Usage

```ts
import { duplicate } from "@sudoo/duplicate";

const variable = {
    //...
}; // something you love
const result = duplicate(variable); // love doubled
```
